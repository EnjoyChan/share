webpack操作

  • 命令行方式，默认读取`webpack.config.js`
  • nodejs api
                      
                        //webpack.config.js
                        module.exports = {
                          entry: './main-entry.js',
                          output: {
                            path: __dirname + './dist.js',
                            filename: 'bundle.js'
                          }
                        }

cmd: webpack

                      
                        //nodejs api
                        var webpack = require('webpack');
                        webpack({
                          entry: './main-entry.js',

                          output: {
                            path: __dirname + './dist.js',
                            filename: 'bundle.js'
                          }
                        }, calllback)
